\text{CMC}_{\text{Lateral}}(r) = 
\frac
{
V(r)
}
{
A(r) \cdot d(r)
}

\text{CMC}_{\text{Bilateral}}(r) = \frac{
V(r_l) + V(r_r)
}
{
A(r_l)d(r_l) + A(r_r)d(r_r)
}


\text{CMC}_{\text{Asym1}}(r)
=
\lVert
\text{CMC}_{\text{Lateral}}(r_l)
-
\text{CMC}_{\text{Lateral}}(r_r)
\rVert


\text{CMC}_{\text{Asym2}}(r)
=
\frac{
\text{CMC}_{\text{Lateral}}(r_l)
}{
\text{CMC}_{\text{Lateral}}(r_r)
}
